    <script>
        document.addEventListener('DOMContentLoaded', function() {
            const trackButton = document.getElementById('track-button');
            const refreshButton = document.getElementById('refresh-button');
            const phoneInput = document.getElementById('phone-number');
            const resultsSection = document.getElementById('results-section');
            const loadingSection = document.getElementById('loading-section');

            // Simulate tracking functionality
            trackButton.addEventListener('click', function() {
                const phoneNumber = phoneInput.value.trim();
                
                if (!phoneNumber) {
                    alert('Please enter a phone number');
                    return;
                }
                
                if (!/^\d{5,15}$/.test(phoneNumber)) {
                    alert('Please enter a valid phone number (numbers only, 5-15 digits)');
                    return;
                }

                // Show loading state
                loadingSection.classList.remove('hidden');
                resultsSection.classList.add('hidden');
                
                // Simulate API call delay
                setTimeout(function() {
                    loadingSection.classList.add('hidden');
                    resultsSection.classList.remove('hidden');
                    
                    // Generate random accuracy
                    const accuracies = ['High accuracy (15m)', 'Medium accuracy (50m)', 'Low accuracy (100m)'];
                    document.getElementById('accuracy').textContent = accuracies[Math.floor(Math.random() * accuracies.length)];
                    
                    // Generate random last seen time
                    const times = ['Active now', 'Just now', '1 min ago'];
                    document.getElementById('last-seen').textContent = times[Math.floor(Math.random() * times.length)];
                    
                    // Generate random network info
                    const networks = ['T-Mobile', 'Verizon', 'AT&T', 'Sprint'];
                    const carriers = networks.map(n => n + ' USA');
                    const networkIndex = Math.floor(Math.random() * networks.length);
                    document.getElementById('network').textContent = networks[networkIndex];
                    document.getElementById('carrier').textContent = carriers[networkIndex];
                    
                    const signals = ['Excellent (5 bars)', 'Good (4 bars)', 'Fair (3 bars)', 'Weak (2 bars)'];
                    document.getElementById('signal').textContent = signals[Math.floor(Math.random() * signals.length)];
                    
                }, 3000);
            });

            // Refresh button functionality
            refreshButton.addEventListener('click', function() {
                loadingSection.classList.remove('hidden');
                resultsSection.classList.add('hidden');
                
                setTimeout(function() {
                    loadingSection.classList.add('hidden');
                    resultsSection.classList.remove('hidden');
                    
                    // Update last updated time
                    const now = new Date();
                    const options = { hour: '2-digit', minute: '2-digit', second: '2-digit' };
                    const timeString = now.toLocaleTimeString('en-US', options);
                    
                    document.getElementById('last-updated').textContent = timeString;
                }, 1500);
            });

            // Form submission handling
            phoneInput.addEventListener('keypress', function(e) {
                if (e.key === 'Enter') {
                    trackButton.click();
                }
            });
        });
    </script>
